                    let count = 0;
                    function searchNum(arr, elem) {
                      let left = -1;
                      let right = arr.length;
                      while (right - left > 1) {
                        count += 1;
                        const mid = Math.floor((left + right) / 2);
                    
                        if (arr[mid] === elem) {
                          return mid;
                          count += 1;
                        }
                        if (arr[mid] > elem) {
                          right = mid;
                        } else {
                          left = mid;
                        }
                      }
                    
                      return -1;
                    }
